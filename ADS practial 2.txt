bimeary search

#include <stdio.h>

int main() {
    int a[10], l, h, m, k;

    printf("Enter 10 elements (sorted): ");
    for (int i = 0; i < 10; i++) {
        scanf("%d", &a[i]);
    }

    printf("Enter the search key: ");
    scanf("%d", &k);

    l = 0;
    h = 9;

    while (l <= h) {
        m = (l + h) / 2;

        if (a[m] == k) {
            printf("Element found at index %d\n", m);
            return 0;
        }
        else if (k > a[m]) {
            l = m + 1;
        }
        else {
            h = m - 1;
        }
    }

    printf("Element not found\n");
    return 0;
}
